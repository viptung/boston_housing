def performance_metric2(y_true, y_predict):

问题 8 - 回答： 将训练数据分成通过Kfold（n_）设定的段数，并且每一段都有机会参与训练和验证，充分利用数据集的一种方法，默认情况下是顺序切分。在GridSearchCV里设置参数字典并遍历多种参数组合，通过在交叉验证下设定多个训练集训练出多个模型并求出对应验证集的评分最后求平均，再对比不同参数的平均评分来确定最佳模型参数，再把最优参数用于整个训练模型来拟合模型。cv_results能看到在不同深度下的K折结果的单独评分及平均评分，能观察到随深度变化各分值的走势。如果不使用交叉验证就容易因为偶然的原因划分出的不理想数据集，导致评分偏高或偏低，不能保证客观和准确，不利于对参数进行搜索对比。交叉验证通过数据的分段参与训练和验证，保证数据客观和准确，最后通过数据的平均评分来判断模型参数的优劣。¶

def cross_val_score(X,y,depth=10,cv=3,scoring = performance_metric):
    from sklearn import tree
    sc=range(depth)
    for k in range(depth):
        reg= tree.DecisionTreeRegressor(max_depth=(k+1))
        s=range(cv)
        for i in range(cv):                             
            sp=(y.shape[0]//cv) 
            X_te =X[i*sp:(i+1)*sp] 
            y_te =y[i*sp:(i+1)*sp]
            X_tr =X[0:i*sp].append(X[(i+1)*sp:y.shape[0]])
            y_tr =y[0:i*sp].append(y[(i+1)*sp:y.shape[0]])
            reg.fit(X_tr,y_tr)
            pr=reg.predict(X_te )
            s[i]=scoring(y_te,pr)
        sc[k]=(k,sum(s)/cv)

    return sc

def fit_model2(X, y):
    """  [X,y]"""                               
    score=cross_val_score(X, y,depth=10,cv=3)   
    #
    best_estimator =sorted(score,key=lambda score:score[1]) 
    best = best_estimator[-1][0]+1
    return tree.DecisionTreeRegressor(max_depth=best)
